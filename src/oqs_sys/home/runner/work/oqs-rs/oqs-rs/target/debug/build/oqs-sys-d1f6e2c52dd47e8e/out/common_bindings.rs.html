        unsafe { &(*(::std::ptr::null::<OQS_CPU_EXTENSIONS>())).SSE3_ENABLED as *const _ as usize },
        36usize,
        concat!(
            "Offset of field: ",
            stringify!(OQS_CPU_EXTENSIONS),
            "::",
            stringify!(SSE3_ENABLED)
        )
    );
}
